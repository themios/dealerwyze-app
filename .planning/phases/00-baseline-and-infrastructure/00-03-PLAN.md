import { vi } from 'vitest'
import type { SupabaseClient } from '@supabase/supabase-js'

/**
 * Fixed test org ID used across all unit tests.
 * Never matches a real org — safe to use in mocks without hitting production.
 */
export const TEST_ORG_ID = 'test-org-00000000-0000-0000-0000-000000000000'

/**
 * Returns a typed mock Supabase client scoped to TEST_ORG_ID.
 *
 * Pass tableHandlers to override per-table behavior:
 *   makeTestClient({ customers: { select: vi.fn()... } })
 *
 * Tables not in tableHandlers return default no-op mocks that resolve
 * with { data: null, error: null }.
 *
 * No real Supabase URL or credentials needed.
 */
export function makeTestClient(
  tableHandlers: Record<string, unknown> = {}
): SupabaseClient {
  const from = vi.fn((table: string) => {
    if (tableHandlers[table]) return tableHandlers[table]
    const chain = {
      select: vi.fn().mockReturnThis(),
      eq: vi.fn().mockReturnThis(),
      neq: vi.fn().mockReturnThis(),
      in: vi.fn().mockReturnThis(),
      is: vi.fn().mockReturnThis(),
      order: vi.fn().mockReturnThis(),
      limit: vi.fn().mockReturnThis(),
      range: vi.fn().mockReturnThis(),
      maybeSingle: vi.fn().mockResolvedValue({ data: null, error: null }),
      single: vi.fn().mockResolvedValue({ data: null, error: null }),
      insert: vi.fn().mockResolvedValue({ data: null, error: null }),
      update: vi.fn().mockReturnThis(),
      upsert: vi.fn().mockResolvedValue({ data: null, error: null }),
      delete: vi.fn().mockReturnThis(),
    }
    return chain
  })
  return { from } as unknown as SupabaseClient
}
```

## Service-Role Policy

`createServiceClient()` bypasses Row Level Security. It is permitted only for:

- **Cron jobs** — no user session available; every cron route validates `validateCronAuth()` before querying
- **Storage operations** — URL signing and direct storage manipulation require bypassing RLS
- **Inbound webhooks** — Twilio, Stripe, Gmail, Telegram — no user session; must validate signature/token first
- **Platform admin routes** — must call `requirePlatformSuperAdmin()` before any query
- **Outbound SMS utilities** — `lib/sms/*` runs outside user request context; org_id must be passed explicitly
- **Audit log writes** — `lib/audit/log.ts` uses service client intentionally; this is a documented exception

**Not permitted:**
- API routes that call `requireProfile()` already have a user session — use `createClient()` (RLS-enforced)
- Do not add new `createServiceClient()` usages in org-scoped handlers without a documented exception comment

**`createClientForRequest()` dual behavior:**
This function returns an auth client for normal user sessions and a service-role client for staff impersonation sessions. All 64 call sites carry this dual behavior. Routes using it must include explicit `.eq('org_id', ...)` scoping so impersonation sessions cannot read cross-tenant data.

**Exception process:**
If a new legitimate use is needed, add a comment above the call:
`// service-role: [reason] — see CLAUDE.md service-role policy`
```
