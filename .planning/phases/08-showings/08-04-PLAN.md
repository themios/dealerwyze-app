```typescript
/**
 * Showing reminders — notifies agents via SMS + email ~24 hours before a scheduled showing.
 * Window: 22–26 hours from now (matches appointmentRemindersV2 pattern).
 * Idempotent: reminder_sent_at IS NULL guard prevents double-firing.
 */
import { sendNotificationEmail } from '@/lib/email/notify.ts'
import type { createServiceClient } from '@/lib/supabase/service'

export async function runShowingReminders(
  supabase: ReturnType<typeof createServiceClient>,
): Promise<{ remindersQueued: number }> {
  const windowStart = new Date(Date.now() + 22 * 60 * 60 * 1000).toISOString()
  const windowEnd   = new Date(Date.now() + 26 * 60 * 60 * 1000).toISOString()

  const { data: showings } = await supabase
    .from('showings')
    .select(`
      id, scheduled_at, org_id, listing_id,
      agent:profiles(id, full_name, email),
      listing:vehicles(id, address),
      contact:customers(id, name)
    `)
    .eq('status', 'scheduled')
    .is('reminder_sent_at', null)
    .gte('scheduled_at', windowStart)
    .lte('scheduled_at', windowEnd)
    .limit(100)

  let remindersQueued = 0

  for (const showing of showings ?? []) {
    const agent   = Array.isArray(showing.agent)   ? showing.agent[0]   : showing.agent
    const listing = Array.isArray(showing.listing) ? showing.listing[0] : showing.listing
    const contact = Array.isArray(showing.contact) ? showing.contact[0] : showing.contact

    if (!agent?.email) continue

    const scheduledAt = new Date(showing.scheduled_at).toLocaleString('en-US', {
      weekday: 'long', month: 'long', day: 'numeric',
      hour: 'numeric', minute: '2-digit', timeZone: 'America/Los_Angeles',
    })

    const address     = listing?.address ?? 'Unknown address'
    const buyerName   = contact?.name ?? 'a buyer'

    // Email to agent
    await sendNotificationEmail({
      to:      agent.email,
      subject: `Showing reminder: ${address}`,
      html:    `<p>Hi ${agent.full_name ?? 'Agent'},</p>
                <p>You have a showing scheduled for <strong>${scheduledAt}</strong> at <strong>${address}</strong> with ${buyerName}.</p>
                <p>This is your 24-hour reminder.</p>`,
    }).catch(err => console.error('[showingReminders] email failed:', err))

    // Fetch org SMS settings for agent SMS (best-effort — skip if no Twilio config)
    const { data: settings } = await supabase
      .from('org_settings')
      .select('twilio_phone_number, twilio_agent_phone')
      .eq('org_id', showing.org_id)
      .maybeSingle()

    const agentPhone = settings?.twilio_agent_phone ?? null
    if (agentPhone) {
      const { createTwilioClient } = await import('@/lib/twilio/client')
      const twilio = createTwilioClient()
      await twilio.messages.create({
        from: settings.twilio_phone_number,
        to:   agentPhone,
        body: `Showing reminder: ${address} with ${buyerName} at ${scheduledAt}.`,
      }).catch(err => console.error('[showingReminders] SMS failed:', err))
    }

    // Mark reminder sent
    await supabase
      .from('showings')
      .update({ reminder_sent_at: new Date().toISOString() })
      .eq('id', showing.id)

    remindersQueued++
  }

  console.log(`[check-tasks] showing reminders queued: ${remindersQueued}`)
  return { remindersQueued }
}
```

Note: If `twilio_agent_phone` doesn't exist in org_settings, omit the SMS block and send email only. Check the org_settings schema before referencing `twilio_agent_phone` — fall back to a simpler email-only path if the column doesn't exist. The SMS send is best-effort only.

If `createTwilioClient` doesn't exist at that import path, use whatever the existing SMS send pattern is in the codebase (check lib/twilio/ or lib/sms/). Do NOT create a custom Twilio client from scratch — reuse whatever exists.
